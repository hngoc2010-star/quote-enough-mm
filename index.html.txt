<script>

function format(x){
return x.toLocaleString('en-US')
}

function calc(){

let thickness = parseFloat(document.getElementById("thickness").value)
let exw = parseFloat(document.getElementById("exw").value)
let logistic = parseFloat(document.getElementById("logistic").value)
let usd = parseFloat(document.getElementById("usd").value)

let width = 1220
let length = 2440

let cbm_per_sheet = (thickness * width * length) / 1000000000

let pcs_container = Math.floor(67 / cbm_per_sheet)

let cbmBuying = cbm_per_sheet * pcs_container
let cbmSelling = cbmBuying

let totalEXW = exw * pcs_container

let totalCost = totalEXW + logistic

let sellingPrice = 450

let sellingAmount = Math.round(sellingPrice * usd * cbmSelling)

let profit = Math.round(sellingAmount - totalCost)

let color = profit < 0 ? "red" : "black"

document.getElementById("result").innerHTML =

"WIDTH: 1220 mm<br><br>"+

"LENGTH: 2440 mm<br><br>"+

"PCS / CONTAINER: "+format(pcs_container)+"<br><br>"+

"CBM CONT. BUYING: "+cbmBuying.toFixed(2)+"<br><br>"+

"CBM CONT. SELLING: "+cbmSelling.toFixed(2)+"<br><br>"+

"TOTAL EXW: "+format(totalEXW)+"<br><br>"+

"TOTAL COST: "+format(totalCost)+"<br><br>"+

"SELLING AMOUNT: "+format(sellingAmount)+"<br><br>"+

"<b style='color:"+color+"'>PROFITS: "+format(profit)+"</b>"

}
