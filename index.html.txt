<script>

function format(x){
return x.toLocaleString("en-US")
}

function calc(){

let thickness = parseFloat(document.getElementById("thickness").value)
let bundles = parseFloat(document.getElementById("bundles").value)
let pcs = parseFloat(document.getElementById("pcs").value)
let exw = parseFloat(document.getElementById("exw").value)
let logistic = parseFloat(document.getElementById("logistic").value)
let usd = parseFloat(document.getElementById("usd").value)

if(isNaN(thickness)||isNaN(bundles)||isNaN(pcs)||isNaN(exw)||isNaN(logistic)||isNaN(usd)){
alert("Please fill all fields")
return
}

let width = 1220
let length = 2440

let totalPCS = bundles * pcs

let cbmPerPCS = (thickness * width * length) / 1000000000

let cbmSelling = cbmPerPCS * totalPCS

let totalEXW = exw * totalPCS

let totalCFR = totalEXW + logistic

let sellingPriceCBM = 450

let sellingAmount = Math.round(cbmSelling * sellingPriceCBM * usd)

let profit = Math.round(sellingAmount - totalCFR)

let color = profit < 0 ? "red" : "black"

document.getElementById("result").innerHTML =

"TOTAL PCS: " + format(totalPCS) + "<br><br>" +

"CBM CONT. SELLING: " + cbmSelling.toFixed(2) + "<br><br>" +

"TOTAL CFR COST: " + format(Math.round(totalCFR)) + "<br><br>" +

"SELLING AMOUNT: " + format(sellingAmount) + "<br><br>" +

"<b style='color:"+color+"'>PROFITS: " + format(profit) + "</b>"

}
